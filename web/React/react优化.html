    # 怎么在react里使用throttle ？

    一般能想到的就是在constructor里用throttle或者debounce来包裹函数

    但是如果用到了event，就会发现event.target是null。

    这时候，就需要把内部函数提取出来，用throttle或者debounce包装，而不能把onChange事件包装起来。

    错误代码：

    class ModalSetting extends React.Component {
  constructor(props) {
    super(props);
    this.state = {
      quickCloseConfirm: props.order_setting.orderS,
    };
    this.handleQuickCloseConfirmChange = throttle(
      this.handleQuickCloseConfirmChange,
      500
    );
  }

// 启用/禁用闪电平仓弹窗
  handleQuickCloseConfirmChange = (event) => {
    this.setState(
      {
        quickCloseConfirm: event.target.checked ? 1 : 0,
      },
      () => {
        // todo
        console.log("调用接口");
      }
    );
  };
}

    正确代码：

    class ModalSetting extends React.Component {

	constructor(props) {
    super(props);
    this.state = {
      quickCloseConfirm: 0,
    };
    this.setQuickCloseConfirm = debounce(this.setQuickCloseConfirm, 100);
  }
  
   // 启用/禁用闪电平仓弹窗
  handleQuickCloseConfirmChange = (event) => {
    const quickCloseConfirm = event.target.checked ? 1 : 0;
    this.setQuickCloseConfirm(quickCloseConfirm);
  };

	// 提取出内部的业务
  setQuickCloseConfirm = (val) => {
    this.setState(
      {
        quickCloseConfirm: val,
      },
      () => {
        // todo
        console.log("调用接口");
      }
    );
  };
  
}
