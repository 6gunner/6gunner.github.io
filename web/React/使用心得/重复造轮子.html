    # 怎么实现，点击空白，隐藏下拉框？

    1、给window绑定click事件，判断事件源，来隐藏下拉框

    2、在点击显示按钮的事件上，将事件阻止冒泡。防止事件传递到外部

    3、在组件销毁或者下拉框隐藏时，将事件解绑

    const selectPoperRef = createRef();
  const [showSelect, setShowSelect] = useState(false);

  const toggleSelectItem = (e) => {
    setShowSelect((visible) => !visible);
    // 阻止冒泡
    e.nativeEvent.stopImmediatePropagation();
  };

  useEffect(() => {
    // 处理隐藏下拉框的逻辑
    const sds = (e) => {
      // 利用contains来判断点击事件源是否在区域内
      if (!selectPoperRef.current.contains(e.target) && showSelect) {
        setShowSelect(false);
      }
    };
    if (showSelect) {
      // 绑定事件
      document.addEventListener("click", sds, false);
    }
    return () => {
      // 解绑时间
      document.removeEventListener("click", sds);
    };
  }, [selectPoperRef, showSelect]);
  }
