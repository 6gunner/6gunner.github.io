      # 3.怎么去自定义组件样式，或者怎么去重写组件的样式

      想要自定义一个select组件的样式：

      你需要

      const SelectIcon = (props) => (
  <Iconfont {...props} type="arrowDown" size="20" />

<Select
                  value={this.state.buy_type}
                  ref={this.inputSelectRef}
                  classes={{
                    root: classes.inputSelectGroup_selectRoot,
                    icon: classes.selectIcon,
                    select: classes.inputSelectGroup_select,
                  }}
                  IconComponent={SelectIcon}
                  MenuProps={{
                    disableScrollLock: true,
                    getContentAnchorEl: null, // 关键，否则vertical不生效
                    anchorOrigin: { horizontal: "left", vertical: "bottom" },
                    classes: {
                      paper: classes.commonPaper,
                      list: classes.menuList,
                    },
                  }}
                >
                  <MenuItem className={classes.menuItem} value="limit">
                    限价
                  </MenuItem>
                  <MenuItem className={classes.menuItem} value="market">
                    市价
                  </MenuItem>
                </Select>
