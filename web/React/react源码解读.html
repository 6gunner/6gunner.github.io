# setState方法原理

setState方法支持传入两个参数 newState, callback

我们都知道setState的方法调用后，并不是马上生效，而是等到下次react渲染时，才会去更新。

原因是因为，setState方法内部，其实是将state和callback添加到一个队列里面。

setState(nextState, callback) {
    // 添加异步队列  不是每次都更新
    this.$updater.addCallback(callback);
    this.$updater.addState(nextState);
 }

这个$updater实例是在Component的构造函数里初始化的。

constructor(props, context) {
    this.$updater = new Updater(this); // 传入当前组件实例
    this.$cache = { isMounted: false };
    this.props = props;
    this.state = {};
    this.refs = {};
    this.context = context;
}

然后我们看一下Updater这个Class类。 它的内部维护了2个数组（队列），一个是pendingState，用来处理状态的。一个是pendingCallbacks，用来处理回调函数。

class Updater {
  constructor(instance) {
    this.instance = instance;
    this.pendingStates = []; // 待处理状态数组
    this.pendingCallbacks = []; // 待处理callback
    this.isPending = false;
    this.nextProps = this.nextContext = null;
    this.clearCallbacks = this.clearCallbacks.bind(this);
  }
}

然后这个Updater实例在addState的时候，就是向pendingStates里去存放新的state。并且，判断当前updater是否是pending状态，如果不是的话，就直接执行update操作。

addState(nextState) {
    if (nextState) {
      this.pendingStates.push(nextState);
      // 这里判断是否已经是pending状态，如果是的话，程序会在Component.forceUpdate执行完后，
      // 发现pendingStates的length > 0, 然后再执行一遍emitUpdate逻辑
      if (!this.isPending) {
        this.emitUpdate();
      }
    }
  }

emitUpdate虽然接收了两个参数，但通过setState方法触发的update调用，都是没有参数的，所以程序会去判断updateQueue这个队列是不是pending的状态。 而props变化引起的update，是会立即触发组件updateComponent。

 emitUpdate(nextProps, nextContext) {
    this.nextProps = nextProps;
    this.nextContext = nextContext;
    // receive nextProps!! should update immediately
    nextProps || !updateQueue.isPending
      ? this.updateComponent()
      : updateQueue.add(this);
  }

updateQueue这个队列，里面维护了多个updater对象，而每个updater对象里存放了很多需要更新的state对象。这里，我们可以得出结论，react的更新都是==批量一次性进行的==。所以，这是为什么多次setState只触发一次更新。

export let updateQueue = {
  updaters: [],
  isPending: false,
  add(updater) {
    this.updaters.push(updater);
  },
  batchUpdate() {
    if (this.isPending) {
      return;
    }
    this.isPending = true;
    let { updaters } = this;
    let updater;
    // toknow why 这里用pop
    while ((updater = updaters.pop())) {
      updater.updateComponent();
    }
    this.isPending = false;
  },
};

updater对象进行updateComponent的逻辑其实很简单。

之前我们知道，通过setStatec触发的更新，nextProps和nextContext对象都是null。而getState()方法，其实就是把pendingStates里的state，和当前的state做一次合并。

所以我们重点要看的是==shouldUpdate==方法的判断。

updateComponent() {
    let { instance, pendingStates, nextProps, nextContext } = this;
    if (nextProps || pendingStates.length > 0) {
      nextProps = nextProps || instance.props;
      nextContext = nextContext || instance.context;
      this.nextProps = this.nextContext = null;
      debugger;
      console.log(this.nextProps === nextProps);
      // getState 合并所有的state的数据，一次更新
      shouldUpdate(
        instance,
        nextProps,
        this.getState(),
        nextContext,
        this.clearCallbacks
      );
    }
  }

getState() {
    let { instance, pendingStates } = this;
    let { state, props } = instance;
    if (pendingStates.length) {
      state = { ...state };
      pendingStates.forEach((nextState) => {
        // 值得注意一下，如果setState传入的是一个数组，那么state会直接设置为数组的第一个元素
        let isReplace = _.isArr(nextState);
        if (isReplace) {
          nextState = nextState[0];
        }
        if (_.isFn(nextState)) {
          nextState = nextState.call(instance, state, props);
        }
        // replace state
        if (isReplace) {
          state = { ...nextState };
        } else {
          state = { ...state, ...nextState };
        }
      });
      // 我们把pendingStates数组置空，这样，pendingStates就可以去存储新的state状态了。
      pendingStates.length = 0;
    }
    return state;
  }

在shouldUpdate逻辑里，我们用到了组件的shouldComponentUpdate生命周期函数。所以这里是优化组件的一个点。

如果组件需要更新，我们在component.$cache对象里存入需要更新的nextProps, nextState, nextContext等。然后执行component.forceUpdate方法


function shouldUpdate(component, nextProps, nextState, nextContext, callback) {
  // 是否应该更新 判断shouldComponentUpdate生命周期
  let shouldComponentUpdate = true;
  if (component.shouldComponentUpdate) {
    shouldComponentUpdate = component.shouldComponentUpdate(
      nextProps,
      nextState,
      nextContext
    );
  }
  if (shouldComponentUpdate === false) {
    component.props = nextProps;
    component.state = nextState;
    component.context = nextContext || {};
    return;
  }
  let cache = component.$cache;
  cache.props = nextProps;
  cache.state = nextState;
  cache.context = nextContext || {};
  component.forceUpdate(callback);
}

forceUpdate做了2件大事

1.首先将我们的component转换成了虚拟vnode

2.将新的vnode和旧的vnode进行一个比较，得到新的dom，并且渲染到页面上

forceUpdate(callback) {
    // 实际更新组件的函数
    let { $updater, $cache, props, state, context } = this;
    if (!$cache.isMounted) {
      return;
    }
    if ($updater.isPending) {
      $updater.addState(state);
      return;
    }
    let nextProps = $cache.props || props;
    let nextState = $cache.state || state;
    let nextContext = $cache.context || context;
    let parentContext = $cache.parentContext;
    let node = $cache.node;
    let vnode = $cache.vnode;
    // 缓存
    $cache.props = $cache.state = $cache.context = null;
    debugger;
    $updater.isPending = true;
    if (this.componentWillUpdate) {
      this.componentWillUpdate(nextProps, nextState, nextContext);
    }
    this.state = nextState;
    this.props = nextProps;
    this.context = nextContext;

    // 下面才是重点  对比vnode
    let newVnode = renderComponent(this);
    let newNode = compareTwoVnodes(
      vnode,
      newVnode,
      node,
      getChildContext(this, parentContext)
    );
    if (newNode !== node) {
      newNode.cache = newNode.cache || {};
      syncCache(newNode.cache, node.cache, newNode);
    }
    $cache.vnode = newVnode;
    $cache.node = newNode;
    // 清除pending 执行didmount生命周期
    clearPending();
    if (this.componentDidUpdate) {
      this.componentDidUpdate(props, state, context);
    }
    if (callback) {
      callback.call(this);
    }
    $updater.isPending = false;
    $updater.emitUpdate();
    // 更新
}

renderComponent其实就是调用了component自身的render方法，返回了vnode.

compareTwoVnodes里面做了diff算法的判断，这个要重点学习。

compareTwoVnodes主要比较新旧两个vnode，然后创建出patch(补丁)对象，描述两个vnode的区别。

export function compareTwoVnodes(vnode, newVnode, node, parentContext) {
    let newNode = node
    if (newVnode == null) {
        // remove
        destroyVnode(vnode, node)
        node.parentNode.removeChild(node)
    } else if (vnode.type !== newVnode.type || vnode.key !== newVnode.key) {
        // replace
        destroyVnode(vnode, node)
        newNode = initVnode(newVnode, parentContext)
        node.parentNode.replaceChild(newNode, node)
    } else if (vnode !== newVnode || parentContext) {
        // same type and same key -> update
        newNode = updateVnode(vnode, newVnode, node, parentContext)
    }
    return newNode
}

diff的比较策略

1.只比较同级的节点，不会进行跨级比较。因为web ui里dom节点跨层级的移动比较少，这样可以提升diff的效率。

2.先判断类型，再判断属性。不同的节点type，生成的树结构肯定不同，直接替换。

3.同一层级的一组节点，有key的话，通过key来比较。
