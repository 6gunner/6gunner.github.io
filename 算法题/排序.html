    # 冒泡排序

    1、外层 for 循环控制循环次数

    2、内层 for 循环进行两数交换，找每次的最大数，排到最后

    3、设置一个标志位，减少不必要的循环

    function bubbleSort(sds) {
for (let i = 0; i < sds.length - 1; i++) {
    for (let j = 0; j < sds.length - 1 - i; j++) {
      if (sds[j] > sds[j + 1] && sds[j] != "null" && sds[j + 1] != "null") {
        let temp = sds[j];
        sds[j] = sds[j + 1];
        sds[j + 1] = temp;
      }
    }
  }
}
